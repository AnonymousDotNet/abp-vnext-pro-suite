namespace {{ context.Project.NameSpace }};

public static class xxDbContextModelCreatingExtensions
{
   public static void ConfigureRMS(this ModelBuilder builder)
   {
         Check.NotNull(builder, nameof(builder));
         builder.Entity<Warehouse>(b =>
                 {
                     b.ToTable(xx.DbTablePrefix + nameof(Warehouse).Pluralize(), xx.DbSchema);
                     b.Property(e => e.Code).IsRequired().HasMaxLength(WarehouseMaxLength.Code).HasComment("仓库编码");
                     b.Property(e => e.Name).IsRequired().HasMaxLength(WarehouseMaxLength.Name).HasComment("仓库名称");
                     b.Property(e => e.Empty).IsRequired().HasComment("实仓还是虚仓 false 虚仓");
                     b.Property(e => e.Dimensions).HasMaxLength(WarehouseMaxLength.Dimensions).HasComment("面积");
                     b.Property(e => e.Province).IsRequired().HasMaxLength(WarehouseMaxLength.Province).HasComment("省");
                     b.Property(e => e.City).IsRequired().HasMaxLength(WarehouseMaxLength.City).HasComment("市");
                     b.Property(e => e.Area).IsRequired().HasMaxLength(WarehouseMaxLength.Area).HasComment("区");
                     b.Property(e => e.Address).IsRequired().HasMaxLength(WarehouseMaxLength.Address).HasComment("详细地址");
                     b.Property(e => e.ContactName).IsRequired().HasMaxLength(WarehouseMaxLength.ContactName).HasComment("联系人");
                     b.Property(e => e.ContactPhone).IsRequired().HasMaxLength(WarehouseMaxLength.ContactPhone).HasComment("联系人电话");
                     b.Property(e => e.Type).IsRequired().HasMaxLength(WarehouseMaxLength.Type).HasComment("仓库类型:保税,VMI,电商,普通,协同");
                     b.HasIndex(e => e.Code).IsUnique();
                     b.ConfigureByConvention();
                 });
   }
}
